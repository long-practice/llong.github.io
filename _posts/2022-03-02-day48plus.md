```python
# 데이터 준비
from __future__ import unicode_literals, print_function, division
from io import open
import glob
import os

def findFiles(path): return glob.glob(path)

print(findFiles('data/*.txt'))

import unicodedata
import string

all_letters = string.ascii_letters + " .,;'"
n_letters = len(all_letters)

# 유니코드 문자열을 ASCII로 변환
def unicodeToAscii(s):
    return ''.join(
      c for c in unicodedata.normalize('NFD', s)
      if unicodedata.category(c) != 'Mn'
      and c in all_letters
    )

# 각 언어의 이름 목록인 category_lines 사전 생성
category_lines = {}
all_categories = []

# 파일을 읽고 줄 단위로 분리
def readLines(filename):
    lines = open(filename, encoding = 'utf-8').read().strip().split('\n')
    return [unicodeToAscii(line) for line in lines]

for filename in findFiles('data/*.txt'):
    category = os.path.splitext(os.path.basename(filename))[0]
    all_categories.append(category)
    lines = readLines(filename)
    category_lines[category] = lines

n_categories = len(all_categories)
print(n_categories)
```

### one-hot vector 변환


```python
# 이름을 Tensor로 변경
# 하나의 문자를 표현하기 위해 크기가 1 x n_letters인 one-hot vector 사용
# one-hot vector는  현재 문자의 주소에만 1값을 가지고 그 외에 나머지는 0으로 채워줌
import torch

# 문자 주소 찾기 ex) 'a' = 0
def letterToIndex(letter):
    return all_letters.find(letter)

# 검증을 위해서 한 개 문자를 Tensor로 변환
def letterToTensor(letter):
    tensor = torch.zeros(1, n_letters)
    tensor[0][letterToIndex(letter)] = 1
    return tensor

# 한 개의 단어를 Tensor로 변환
def lineToTensor(line):
    tensor = torch.zeros(len(line), 1, n_letters)
    for li, letter in enumerate(line):
        tensor[li][0][letterToIndex(letter)] = 1
    return tensor

print(letterToTensor('J'))
print(lineToTensor('Jones').size())
```

### 네트워크 구성


```python
import torch.nn as nn

class RNN(nn.Module):
    def __init__(self, input_size, hidden_size, output_size):
        super(RNN, self).__init__()

        self.hidden_size = hidden_size

        self.i2h = nn.Linear(input_size + hidden_size, hidden_size)
        self.i2o = nn.Linear(input_size + hidden_size, output_size)
        self.softmax = nn.LogSoftmax(dim = 1)

  
    def forward(self, input, hidden):
        combined = torch.cat((input, hidden), 1)
        hidden = self.i2h(combined)
        output = self.i2o(combined)
        output = self.softmax(output)
        return output, hidden

    def initHidden(self):
        return torch.zeros(1, self.hidden_size)

n_hidden = 128
rnn = RNN(n_letters, n_hidden, n_categories)
```

```python
# 네트워크 학습
# RNN의 마지막 층은 nn.logSoftmax 이므로 nn.NLLLoss 사용

# 오차 함수
criterion = nn.NLLLoss().cuda()
learning_rate = 0.005

# RNN 네트워크 생성
rnn = RNN(n_letters, n_hidden, n_categories)
rnn = rnn.cuda()

def train(category_tensor, line_tensor):
    # 은닉층 초기화
    hidden = rnn.initHidden().cuda()
    
    # 그레디언트 초기화
    rnn.zero_grad()
    
    # 출력, 은닉 상태
    for i in range(line_tensor.size()[0]):
        output, hidden = rnn(line_tensor[i].cuda(), hidden)
    
    # 오차 계산, 역전파 연산
    loss = criterion(output, category_tensor)
    loss.backward()
    
    # 매개변수의 경사도에 학습률을 곱해서 그 매개변수 값에 더함
    for p in rnn.parameters():
        p.data.add_(p.grad.data, alpha = -learning_rate)
    
    return output, loss.item()
```

### 평가
- 모든 실제 언어가 네트워크에서 어떤 언어로 추출되는지 나타내는 혼란 행렬(confusion matrix)을 생성
- 주축에서 벗어난 밝은 점을 선택하여 잘못 추측한 언어를 표시


```python
from matplotlib.pyplot as plt

confusion = torch.zeros(n_categories, n_categories)
```
